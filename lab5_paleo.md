lmp <- function(model) {
  if (class(model)!="lm") {
    stop("Not an object of class 'lm'")
    }
  f <- summary(model)$fstatistic
  p <- pf(f[1], f[2], f[3], lower.tail=FALSE)
  attributes(p) <- NULL
  return(p)
  }
